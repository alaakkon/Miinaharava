5   /**
6    *
7    * Luokka tarjoaa Miinaharava pelin pelin kulkemisen toiminnat.
8    */
9   public class PelinKulku {
10  
11      private Lauta lauta;
12  
13      public PelinKulku(int korkeus, int leveys) {
14          this.lauta = new Lauta(korkeus, leveys);
15          lauta.alustaLauta();
16      }
17  
18      /**
19       * Metodi hakee Lauta luokan pelilaudan.
20       */
21      public Lauta haeLauta() {
22          return lauta;
23      }
24  
25      /**
26       * Metodi tarkistaa sisältääkö annettujen parametrien sisältämä ruutu
27       * miinan.
28       *
29       * @param y korkeuden koordinaatti
30       * @param x leveyden koordinaatti
31       *
32       */
33      public boolean onMiina(int y, int x) {
34          if (lauta.haeRuutuTaulukko()[y][x].haeTila() == 9) {
35              return true;
36          } else {
37              return false;
38          }
39      }
40  
41      /**
42       * Metodi toimittaa käyttäjän syötteen mukaisen pelinkulun.
43       *
44       * @param y korkeuden koordinaatti
45       * @param x leveyden koordinaatti
46       */
47      public void pelaa(int y, int x) {
48          if (onMiina(y, x)) {
49              System.out.println("Hävisit!!!");
50          } else if (lauta.haeRuutuTaulukko()[y][x].haeTila() != 0) {
51              avaaRuutu(y, x);
52          } else {
53              avaaNolla(y, x);
54          }
55      }
56  
57      /**
58       * Metodi avaa syötteen mukaisen ruudun.
59       *
60       * @param y korkeuden koordinaatti
61       * @param x leveyden koordinaatti
62       */
63      public void avaaRuutu(int y, int x) {
64  
65          lauta.haeRuutuTaulukko()[y][x].muutaAvoimuus(true);
66      }
67  
68      /**
69       * Metodi avaa ruudun, joka ei ole kosketuksissa miinoja sisältäviin
70       * ruutuihin, sekä kaikki ruudun kosketuksissa olevat ruudut ja näitä
71       * seuraavat ruudut,jotka eivät myöskään ole kosketuksissa miinoja
72       * sisältäviin ruutuihin.
73       *
74       * @param y korkeuden koordinaatti
75       * @param x leveyden koordinaatti
76       */
77      public void avaaNolla(int y, int x) {
78          lauta.haeRuutuTaulukko()[y][x].muutaAvoimuus(true);
79          avaaKaikkiViereisetNollat(y, x);
80      }
81  
82      /**
83       * Avataan kaikki ruudut, jotka ovat tyhjiä ja kosketuksissa arvattuun,
84       * tyhjään ruutuun.
85       *
86       * @param b korkeuden koordinaatti
87       * @param a leveyden koordinaatti
88       */
89      public void avaaKaikkiViereisetNollat(int b, int a) {
90          int xAlku = Math.max(a - 1, 0);
91          int xLoppu = Math.min(a + 1, lauta.haeRuutuTaulukko()[0].length - 1);
92          int yAlku = Math.max(b - 1, 0);
93          int yLoppu = Math.min(b + 1, lauta.haeRuutuTaulukko().length - 1);
94          for (int i = yAlku; i <= yLoppu; i++) {
95              for (int j = xAlku; j <= xLoppu; j++) {
96                  if (lauta.haeRuutuTaulukko()[i][j].haeTila() == 0 && !lauta.haeRuutuTaulukko()[i][j].onAuki()) {
97                      avaaNolla(i, j);
98                  }
99              }
100         }
101 
102     }
